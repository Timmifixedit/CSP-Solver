21  /**
22  * Recursive backtracking search for csp::Csps. Prefer using the wrapper function csp::solve
23  * @tparam VarPtr Pointer-type to a type derived from csp::Variable
24  * @tparam Strategy Type of value selection strategy during search
25  * @param problem CSP to be solved
26  * @param strategy value selection strategy object used during during search
27  * @return True if problem was solved, false otherwise
28  */
29  template<typename VarPtr, typename Strategy>
30  bool recursiveSolve(Csp<VarPtr> &problem, const Strategy &strategy) {
31  using Domain = typename Csp<VarPtr>::VarT::DomainT;
32  VarPtr nextVar = strategy(problem);
33  if (nextVar->isAssigned()) {
34  return true;
35  }
36 
37  //Moving storage as it will be overwritten by assign() anyway
38  Domain valueDomain = std::move(nextVar->valueDomain());
39  for (auto &val : valueDomain) {
40  nextVar->assign(std::move(val));
41  // Back up value domains of all variables. This is considerably faster than creating a deep copy of the
42  // whole Csp
43  auto cp = util::makeCspCheckpoint(problem);
44  if (!util::ac3(problem)) {
45  util::restoreCspFromCheckpoint(problem, cp);
46  continue;
47  }
48 
49  if (recursiveSolve(problem, strategy)) {
50  return true;
51  }
52 
53  util::restoreCspFromCheckpoint(problem, cp);
54  }
55 
56  return false;
57  }

59  /**
60  * Solves a CSP. If a solution exists, the value domains of each variable in the given Csp will be reduced to
61  * exactly one value. If multiple solutions exist, it is unspecified which is found. If no solution exists, false
62  * is returned but the value domains of the variables might still be altered
63  * @tparam VarPtr Pointer-type to a type derived from csp::Variable
64  * @tparam Strategy Type of value selection strategy during search (default: minimum remaining values strategy).
65  * Has to provide ()-Operator and return VarPtr from given csp::Csp
66  * @param problem CSP to be solved
67  * @return True if problem was solved, false otherwise
68  */
69  template<typename VarPtr, typename Strategy = strategies::Mrv<VarPtr>>
70  bool solve(Csp<VarPtr> &problem, const Strategy &strategy = Strategy()) {
71  static_assert(std::is_invocable_r_v<VarPtr, Strategy, Csp<VarPtr>>,
72  "Invalid strategy object! Must map from csp::Csp -> VarPtr");
73  if (std::empty(problem.variables)) {
74  return true;
75  }
76 
77  if (!util::ac3(problem)) {
78  return false;
79  }
80 
81  return recursiveSolve(problem, strategy);
82  }
