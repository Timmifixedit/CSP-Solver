21  /**
22  * Removes all inconsistent values from the source node of the given csp::Arc. As a result, the source node's
23  * value domain only contains values for which a valid value in the domain of the destination node exists.
24  * @tparam VarPtr Pointer-type to a type derived from csp::Variable
25  * @param arc Arc to be processed
26  * @return True if the value domain of the source node was modified, false otherwise
27  */
28  template<typename VarPtr>
29  bool removeInconsistent(const Arc<VarPtr> &arc) {
30  bool removed = false;
31  VarPtr from = arc.from();
32  VarPtr to = arc.to();
33  for (auto it = from->valueDomain().begin(); it != from->valueDomain().end();) {
34  bool consistent = false;
35  for (const auto &val : to->valueDomain()) {
36  if (arc.constraintSatisfied(*it, val)) {
37  consistent = true;
38  break;
39  }
40  }
41 
42  if (!consistent) {
43  it = from->valueDomain().erase(it);
44  removed = true;
45  } else {
46  ++it;
47  }
48  }
49 
50  return removed;
51  }

53  /**
54  * Obtains arc consistency in a CSP using the AC3-algorithm
55  * @tparam VarPtr Pointer-type to a type derived from csp::Variable
56  * @param problem The CSP to be processed
57  * @return True if arc consistency was obtained, false if not possible
58  */
59  template<typename VarPtr>
60  bool ac3(Csp<VarPtr> &problem) {
61  using ArcT = typename Csp<VarPtr>::ArcT;
62  typename Csp<VarPtr>::ArcListT arcs = problem.arcs;
63  while (!arcs.empty()) {
64  const ArcT current = arcs.front();
65  arcs.pop_front();
66  if(removeInconsistent(current)) {
67  if (current.from()->valueDomain().empty()) {
68  return false;
69  }
70 
71  auto it = problem.incomingNeighbours.find(current.from());
72  assert(it != problem.incomingNeighbours.end());
73  std::copy_if(it->second.begin(), it->second.end(), std::back_inserter(arcs),
74  [&current](const auto &arc) {return arc.from() != current.to();});
75  }
76  }
77 
78  return true;
79  }

84  /**
85  * Backs up all value domains of all variables in a CSP
86  * @tparam VarPtr Pointer-type to a type derived from csp::Variable
87  * @param problem The CSP to be backed up
88  * @return vector of csp::Variable domains. Domains are ordered according to the variables in the CSP
89  */
90  template<typename VarPtr>
91  auto makeCspCheckpoint(const Csp<VarPtr> &problem) -> CspCheckpoint<typename Csp<VarPtr>::VarT> {
92  using VarType = typename Csp<VarPtr>::VarT;
93  CspCheckpoint<VarType> ret;
94  ret.reserve(problem.variables.size());
95  for (const auto &var : problem.variables) {
96  ret.emplace_back(var->valueDomain());
97  }
98 
99  return ret;
