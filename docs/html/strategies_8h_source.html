25  /**
26  * Minimum remaining values strategy for variable selection during search. Chooses the variable with the fewest
27  * remaining possible values
28  * @tparam VarPtr Pointer-type to a type derived from csp::Variable
29  */
30  template<typename VarPtr>
31  struct Mrv {
32  VarPtr operator() (const Csp<VarPtr> &problem) const {
33  auto it = std::min_element(problem.variables.begin(), problem.variables.end(),
34  [](const auto &lhs, const auto &rhs) {
35  return rhs->isAssigned() || (!lhs->isAssigned() &&
36  lhs->valueDomain().size() < rhs->valueDomain().size());
37  });
38 
39  assert(it != problem.variables.end());
40  return *it;
41  }
42  };
