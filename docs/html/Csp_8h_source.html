134  * @return csp::Csp representing the problem induced by the given variables and constraints
135  * @note When using csp::Constraints to specify the constraints, specify them only once. A csp::Constraint for
136  * e.g. A < B fully represents the constraint between the csp::Variable A and B. Specifying A < B and B > A
137  * may lead to performance loss during search!
138  */
139  template<typename VarIt, typename ConstrIt, std::enable_if_t<type_traits::is_constraint<
140  std::remove_reference_t<decltype(*std::declval<ConstrIt>())>>::value, int> = 0>
141  auto make_csp(VarIt vBegin, VarIt vEnd, ConstrIt cBegin, ConstrIt cEnd)
142  -> Csp<std::decay_t<decltype(*++cBegin, cBegin == cEnd, ++vBegin, vBegin == vEnd, *vBegin)>> {
143  using VarPtr = std::decay_t<decltype(*vBegin)>;
144  using ArcT = typename Csp<VarPtr>::ArcT;
145  std::vector<ArcT> arcs;
146  while (cBegin != cEnd) {
147  auto[normal, reverse] = cBegin->getArcs();
148  arcs.emplace_back(std::move(normal));
149  arcs.emplace_back(std::move(reverse));
150  ++cBegin;
151  }
152 
153  return make_csp(vBegin, vEnd, std::begin(arcs), std::end(arcs));
